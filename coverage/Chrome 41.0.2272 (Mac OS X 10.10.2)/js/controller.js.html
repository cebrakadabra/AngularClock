clockapp.controller('MainController', ['$scope', '$location', function($scope, $location){
	
	// goTo wird benutzt um zwischen den Routingviews
	// leichter hin und her navigieren zu können
	$scope.goTo = function ( path ) {
      $location.path( path );
    };
 
}]);
